function createVariables() {
    variables_container.innerHTML = '';
    document.querySelectorAll('.vari').forEach((v, index) => {
        if (v.classList.contains('code')) {
            const name = v.querySelector('input[type="text"]');
            const value = v.querySelector('input[type="number"]');
            const new_variable = document.createElement('div');
            const n = document.createElement('h');
            const new_value = document.createElement('p');
            new_variable.className = 'created_variables';
            variables_container.appendChild(new_variable);
            n.innerText = name.value;
            new_variable.appendChild(n);
            new_value.innerText = value.value;
            new_value.style.display = "none";
            new_variable.style.cursor = "grab";
            new_variable.appendChild(new_value);
            new_variable.setAttribute('draggable', 'true')
            new_variable.id = `draggable-${Date.now()}-${index}`;

            new_variable.addEventListener('dragstart', (e) => {
                e.dataTransfer.setData('text/plain', e.target.id);
                isCreatedVariable = true;
            });
        }
    });
}
